pipeline{

	agent any

	environment {
		DOCKERHUB_CREDENTIALS=credentials('kartikeya532001')
	}

	stages {
	   
		stage('setup') {

			steps {
				sh 'npm install'
			}
		}
		
		stage('Build') {

			steps {
				sh 'docker build -t kartikeya532001/nodeapp_test:1.0 .'
			}
		}

		stage('Login') {

			steps {
				sh 'echo $DOCKERHUB_CREDENTIALS_PSW | docker login -u $DOCKERHUB_CREDENTIALS_USR --password-stdin'
			}
		}

		stage('Push') {

			steps {
				sh 'docker push kartikeya532001/nodeapp_test:1.0'
			}
		}
		stage('deployment') {
		steps {
			
			sh 'sudo minikube delete'
			sh 'minikube start'
			sh 'eval $(minikube docker-env)'
			sh 'docker pull kartikeya532001/nodeapp_test:1.0'
			sh 'sudo export KUBECONFIG=/home/kartikeya/.kube/config'
			sh 'sudo kubectl apply -f deploymentservice.yml'
			
			
		}
		}
	
	
	}

	

}
